# Step 1: create a fresh randomly-initialised .nnue (no source .nnue needed)
#   --init-nnue  creates FC/FT/PSQT from N(mean,std) measured from SF15.1
#   --write-nnue writes the result (required with --init-nnue)
perl comp.pl init_nnue NNUE=1 TDLEAF=1
./EXchess_vinit_nnue --init-nnue --write-nnue nn-zeroed.nnue

# Step 2: build two training versions pointing to the new net
perl comp.pl train_zeroed NNUE=1 NNUE_NET=nn-zeroed.nnue TDLEAF=1
perl comp.pl train_zeroed_ro NNUE=1 NNUE_NET=nn-zeroed.nnue TDLEAF=1 TDLEAF_READONLY=1

# Third, create a version to test the zeroed nune file..
perl comp.pl test_zeroed NNUE=1 NNUE_NET=nn-zeroed.nnue

# Third, start training
#
# NOTE: we are playing each version against itself, including the trained weights to that point 
#  Each new game in the set updates the train_zeroed but not train_zeroed_ro version.
#
python3 tdleaf_selfplay.py EXchess_vtrain_zeroed -n 5000 --depth 6 --engine2 EXchess_vtrain_zeroed_ro --depth2 6 --move-timeout 10 

# Four, examine the learning compared to the original
python3 compare_fc_weights.py nn-zeroed.nnue nn-zeroed.tdleaf.bin --ft-weights
